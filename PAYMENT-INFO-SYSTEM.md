#### API ใหม่:

**1. `GET /api/payment/config` - ดึงข้อมูลการชำระเงิน (Public - ทุกคนดูได้)**
```javascript
app.get('/api/payment/config', (req, res) => {
    const config = readJSON(PAYMENT_CONFIG_FILE);
    res.json({ success: true, config });
});
```

**2. `POST /api/admin/payment/config` - อัพเดตข้อมูลการชำระเงิน (DEV เท่านั้น)**
```javascript
app.post('/api/admin/payment/config', (req, res) => {
    const { adminUsername, truemoney, bank } = req.body;
    
    // Check DEV only
    const admin = userData?.users.find(u => u.username === adminUsername);
    if (!admin || (admin.role !== 'admin' && admin.role !== 'dev')) {
        return res.json({ success: false, message: 'ไม่มีสิทธิ์เข้าถึง (เฉพาะ DEV)' });
    }
    
    // Update config
    if (truemoney) config.truemoney = { ...config.truemoney, ...truemoney };
    if (bank) config.bank = { ...config.bank, ...bank };
    
    writeJSON(PAYMENT_CONFIG_FILE, config);
    res.json({ success: true, message: 'อัพเดตข้อมูลสำเร็จ', config });
});
```

**3. `POST /api/admin/payment/upload-qr` - อัพโหลด QR Code (DEV เท่านั้น)**
```javascript
app.post('/api/admin/payment/upload-qr', upload.single('qrImage'), (req, res) => {
    const { adminUsername, paymentType } = req.body; // 'truemoney' or 'bank'
    
    // Check DEV only
    const admin = userData?.users.find(u => u.username === adminUsername);
    if (!admin || (admin.role !== 'admin' && admin.role !== 'dev')) {
        return res.json({ success: false, message: 'ไม่มีสิทธิ์เข้าถึง (เฉพาะ DEV)' });
    }
    
    const qrCodeUrl = `/uploads/${req.file.filename}`;
    config[paymentType].qrCodeUrl = qrCodeUrl;
    
    writeJSON(PAYMENT_CONFIG_FILE, config);
    res.json({ success: true, message: 'อัพโหลด QR Code สำเร็จ', qrCodeUrl });
});
```

async function updateTruemoneyConfig() {
    const accountNumber = document.getElementById('truemoneyNumber').value.trim();
    const accountName = document.getElementById('truemoneyName').value.trim();
    const qrFile = document.getElementById('truemoneyQR').files[0];
    
    // Validate
    if (!accountNumber || accountNumber.length !== 10) {
        alert('เลขบัญชีต้องมี 10 หลัก');
        return;
    }
    
    // Update text data
    const res = await fetch('/api/admin/payment/config', {
        method: 'POST',
        body: JSON.stringify({ adminUsername, truemoney: { accountNumber, accountName } })
    });
    
    // Upload QR if exists
    if (qrFile) {
        const formData = new FormData();
        formData.append('qrImage', qrFile);
        formData.append('adminUsername', adminUsername);
        formData.append('paymentType', 'truemoney');
        
        await fetch('/api/admin/payment/upload-qr', { method: 'POST', body: formData });
    }
    
    showPaymentMessage('บันทึก TrueMoney สำเร็จ!', 'success');
    loadPaymentConfig();
}
```

function showPaymentMessage(message, type) {
    const msgEl = document.getElementById('paymentMessage');
    msgEl.style.display = 'block';
    msgEl.textContent = message;
    
    if (type === 'success') {
        msgEl.style.background = 'rgba(46,204,113,0.2)';
        msgEl.style.color = '#2ecc71';
    } else if (type === 'error') {
        msgEl.style.background = 'rgba(255,71,87,0.2)';
        msgEl.style.color = '#ff4757';
    }
    
    setTimeout(() => msgEl.style.display = 'none', 5000);
}

   ↓
2. เรียก loadPaymentConfig()
   ↓
3. GET /api/payment/config
   ↓
4. แสดง TrueMoney Card + Bank Card
   ↓
5. แสดงเลขบัญชี + ชื่อบัญชี + QR Code (ถ้ามี)
```
