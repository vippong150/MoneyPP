```html
    ...

```

#### ฟังก์ชัน `loadActiveBorrows()` - เพิ่มส่วนโหลด pending

```javascript
async function loadActiveBorrows() {
    const res = await fetch('/api/borrow/history', {...});
    const data = await res.json();
    
    if (data.success) {
        // 1. โหลดรายการรออนุมัติ (ใหม่)
        const pending = data.history.filter(b => b.status === 'pending');
        const pendingSection = document.getElementById('pendingBorrowsSection');
        
        if (pending.length > 0) {
            pendingSection.innerHTML = `
                <div style="background:linear-gradient(...);border:2px solid #ffc107;">
                    <div>
                        <i class="fas fa-hourglass-half"></i> 
                        รายการยืมเงินรออนุมัติ (${pending.length})
                    </div>
                    ${pending.map(borrow => {
                        const borrowDate = new Date(borrow.date);
                        const dateStr = borrowDate.toLocaleDateString('th-TH', {
                            year: 'numeric',
                            month: 'long',
                            day: 'numeric'
                        });
                        const timeStr = borrowDate.toLocaleTimeString('th-TH', {
                            hour: '2-digit',
                            minute: '2-digit'
                        });
                        
                        return `
                            <div style="background:rgba(0,0,0,0.2);padding:14px;">
                                <div style="display:flex;justify-content:space-between;">
                                    <div>
                                        <div>💰 ${borrow.amount.toFixed(2)} บาท</div>
                                        <div>📅 ส่งคำขอเมื่อ: ${dateStr}</div>
                                        <div>⏰ เวลา: ${timeStr}</div>
                                    </div>
                                    <div>
                                        <div>⏳ รออนุมัติ</div>
                                        <div>ID: ${borrow.id}</div>
                                    </div>
                                </div>
                            </div>
                        `;
                    }).join('')}
                </div>
            `;
        } else {
            pendingSection.innerHTML = '';
        }
        
        // 2. โหลดรายการที่อนุมัติแล้ว (เหมือนเดิม)
        const active = data.history.filter(b => 
            b.status === 'approved' || b.status === 'return_pending'
        );
        ...
    }
}
```
